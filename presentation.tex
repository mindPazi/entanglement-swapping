%  NOISE MODEL
% ============================================================
\begin{frame}{Noise Model: Depolarization}
  \begin{columns}[c]
    \begin{column}{0.50\textwidth}
      Quantum memories \textbf{depolarize} while waiting.

      \vskip 0.5em
      {\small Single-qubit depolarizing channel (QuantumSavory convention):}
      \[
        \mathcal{E}(\rho) = (1 - p)\,\rho + p\,\frac{I}{2}
      \]

      \vskip 0.2em
      {\small Noise tied to waiting time:}
      \[
        p(\Delta t) = 1 - e^{-\Delta t / \tau}
      \]

      \vskip 0.2em
      {\small $\tau = -1/\ln(1 - \pw)$, \; $\pw$ = per-step depolarization rate.}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{block}{Why Time Matters}
        {\small If $\ps < 1$, faster links \textbf{wait} for slower ones.}
        \[
          T = \max_{i=1}^{N+1} T_i, \quad T_i \sim \mathrm{Geom}(\ps)
        \]
        {\small Larger $T$ $\Rightarrow$ more depolarization $\Rightarrow$ lower $F$.}
      \end{block}

      \vskip 0.2em
      \begin{block}{Fidelity}
        \vspace{-0.6em}
        \[ F = \tfrac{1}{4}\!\left(1 + \expval{XX} + \expval{ZZ} - \expval{YY}\right) \]
        \vspace{-0.8em}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Phase 5: Theoretical Framework}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Distribution Time}
        {\small For $N+1$ i.i.d.\ links with $T_i \sim \mathrm{Geom}(\ps)$:}
        \[
          \mathbb{E}[T] = \sum_{t=0}^{\infty} \left[1 - \big(1 - (1-\ps)^t\big)^{N+1}\right]
        \]

        \vskip 0.1em
        {\small Harmonic approximation:}
        \[
          \mathbb{E}[T] \approx \frac{H(N+1)}{\ps}, \quad H(n) = \sum_{k=1}^n \frac{1}{k}
        \]
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Werner Model for Fidelity}
        {\small Each link $i$ waits $\Delta t_i = T - T_i$ time steps.
        Werner parameter per link:}
        \[
          w_i = \left(1 - p(\Delta t_i)\right)^2
        \]
        {\small where $p(\Delta t) = 1 - (1 - \pw)^{\Delta t}$.
        Swapping multiplies Werner parameters:}
        \[
          w_{\mathrm{tot}} = \prod_{i=1}^{N+1} w_i \qquad F = \frac{1 + 3\, w_{\mathrm{tot}}}{4}
        \]
      \end{block}
    \end{column}
  \end{columns}
\end{frame}
